matplotlib.rcParams['figure.figsize'] = (20.0, 10.0)


x = np.linspace(-5,5,1000)
y1 = 2*x + 5

def sigmoid(x):
    return 1/(1+np.exp(-x))

y2 = sigmoid(y1)

f, ax = plt.subplots()
ax.plot(x,y1, color='r', label='y=2x+5')
ax.plot(x,y2, color='b', label='y=sigmoid(2x+5)')
ax.legend()
ax.grid(True)
ax.set_ylim(-1,1)
